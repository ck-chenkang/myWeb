## 匿名插槽

用组件生成一段内容
```html
<body>
    <div id="app">
        <tag1mes></tag1mes>        
    </div>
</body>
<script>
    Vue.component('tag1mes', {
        props: ['post'],
        template: `        
        <dl>
            <dt>狮子</dt>
            <dd>是一种生存在非洲与亚洲的大型猫科动物</dd>
        </dl>
        `
    })

    var vm = new Vue({
        el: '#app',
        data: {}
    })
</script>
```

用插槽，下面的渲染结果和上面的一样
```html
<body>
    <div id="app">
        <tag1mes></tag1mes>        
    </div>
</body>
<script>
    Vue.component('tag1mes', {
        props: ['post'],
        template: `        
        <dl>
            <dt><slot></slot></dt>
            <dd>是一种生存在非洲与亚洲的大型猫科动物</dd>
        </dl>
        `
    })

    var vm = new Vue({
        el: '#app',
        data: {}
    })
</script>
```

查找内部可以有默认的值，这里要是没有老虎两个字就会被渲染成默认的狮子
```html
<body>
    <div id="app">
        <tag1mes>
            老虎
        </tag1mes>        
    </div>
</body>
<script>
    Vue.component('tag1mes', {
        props: ['post'],
        template: `        
        <dl>
            <dt><slot>狮子</slot></dt>
            <dd>是一种生存在非洲与亚洲的大型猫科动物</dd>
        </dl>
        `
    })

    var vm = new Vue({
        el: '#app',
        data: {}
    })
</script>
```

## 具名插槽

给插槽起个名字，这样的话，组件中插槽的内容就和外部有名字的一样了。
`<template slot="girl">漂亮、美丽、购物、逛街</template>` 注意这个语法

```html
<div id="app">
    <child-component>
        <template slot="girl">
            漂亮、美丽、购物、逛街
        </template>
        <template slot="boy">
            帅气、才实
        </template>
        <div>
            我是一类人，
            我是默认的插槽
        </div>
    </child-component>
</div>
<script>
    Vue.component('child-component',{
        template:`
            <div>
            <h4>这个世界不仅有男人和女人</h4>

            <slot name="girl"></slot>

            <div style="height:1px;background-color:red;"></div>

            <slot name="boy"></slot>

            <div style="height:1px;background-color:red;"></div>

            <slot></slot>
            </div>
        `
    })
    let vm = new Vue({
        el:'#app',
        data:{

        }
    })
</script>
```

### 作用域插槽

```html
<div id="app">
    <child>
        <template slot-scope="a">　
            {{a}}
        </template>
    </child>
</div>
<script>
    Vue.component('child',{
        template:`
            <div>
                <slot say="你好"></slot>
            </div>
        `
    })

    let vm = new Vue({
        el:'#app',
        data:{

        }
    })
</script>
```
